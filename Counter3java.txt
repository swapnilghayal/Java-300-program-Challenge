public class Counter3
{
	public static void main(String args[])
	{
		int n=5;
		int counter1=1;
		int counter2=n*(n+1)/2;
		
		for(int i=1; i<=n; i++)
		{
		int c1=counter1;
		int c2=counter2;

		for(int j=n,j1=i; j>=i || j1<=n; j--,j1++)
		{
			System.out.print((char)(c1+64));
			System.out.print((char)(c2+96));
			System.out.print("\t");

		c1+=j;
		c2-=j1;
		}
		
		System.out.println();
		
		counter1++;
		counter2-=i+1;
	

		}

	


	}








}
